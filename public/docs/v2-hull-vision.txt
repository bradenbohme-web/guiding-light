function compileGraph(nodes: NodeSpec[], edges: [NodeId,NodeId][]) {
  const stageOrder: Stage[] = [
    "BASE_FIELDS","PLACEMENT","CONNECTIONS","FEATURES","MESHER_FAST","MESHER_FINAL"
  ];

  const deps = buildAdjacency(edges);

  const byStage = groupBy(nodes, n => n.stage);

  const pipeline: NodeId[] = [];
  for (const st of stageOrder) {
    const sub = byStage.get(st) ?? [];
    const subEdges = restrictEdges(edges, sub);
    const sorted = topoSortStable(sub, subEdges); // stable sort = deterministic
    pipeline.push(...sorted.map(n=>n.id));
  }

  return { pipeline, deps };
}

5.4 Dirty propagation

When a curve knot moves in an ortho part:

mark that part node dirty

propagate forward along deps

rebuild only those nodes, in pipeline order

function markDirty(start: NodeId, deps: Map<NodeId,NodeId[]>) {
  const dirty = new Set<NodeId>();
  const q = [start];
  while (q.length) {
    const n = q.shift()!;
    if (dirty.has(n)) continue;
    dirty.add(n);
    for (const m of deps.get(n) ?? []) q.push(m);
  }
  return dirty;
}

function runPipeline(pipeline: NodeId[], dirty: Set<NodeId>, mode:"FAST"|"FINAL") {
  for (const id of pipeline) {
    if (!dirty.has(id)) continue;
    executeNode(id, mode); // see next section
  }
}

const smoothMin = (a:number,b:number,k:number)=> {
  const h = clamp01(0.5 + 0.5*(b-a)/k);
  return (1-h)*b + h*a - k*h*(1-h);

// A very practical “cone-like” SDF along +X (frustum)
function sdBowCone(p:[number,number,number], x0:number, x1:number, r0:number, r1:number){
  // x0..x1 is bow region; radius shrinks from r0 to r1 as x→x1
  const [x,y,z] = p;
  const t = clamp01((x - x0)/(x1 - x0));
  const r = (1-t)*r0 + t*r1;
  // radial in YZ (you can bias Y vs Z if needed)
  const q = Math.sqrt(y*y + z*z) - r;
  // also clamp “ends” so it doesn't extend infinitely
  const capA = x0 - x;
  const capB = x - x1;
  return Math.max(q, Math.max(capA, capB));
}

// Bevel plane: keeps a ~45° “knife edge” instead of a point
function sdBevelPlane(p:[number,number,number], xTip:number, bevelLen:number, bevelAngleRad:number){
  // Plane normal lives in X-Y (or X-(some composite)) depending on whether bevel slopes down
  // simplest: plane slants down as it goes back from the tip
  const [x,y,_z] = p;
  // line: at x=xTip, allow highest; at x=xTip+bevelLen, plane drops by bevelLen*tan(angle)
  const drop = (x - xTip) * Math.tan(bevelAngleRad);
  // y - (y0 - drop) <= 0 is “inside”; so distance-like:
  return y - (-drop); // choose y0=0 baseline; you’ll offset to match hull bottom line
}

// Final bow cap field (to be blended into hull/deck)
function bowKnifeField(p:[number,number,number], P: any){
  const cone = sdBowCone(p, P.x0, P.x1, P.r0, P.r1);

  // The bevel is a CUT: keep region BELOW the plane (or above, depending on sign)
  const bevel = sdBevelPlane(p, P.xTip, P.bevelLen, P.bevelAngle);

  // Combine: intersection (max) if bevel is a half-space cut
  let d = Math.max(cone, bevel);

  // Round edges globally (approx): just subtract radius from distance field
  d -= P.bowRadiusRound;

  return d;
}

type Keypoints = Record<string, {x:number,z:number, s:number}>;

function computeKeypoints(perim: Vec2[]): Keypoints {
  const P = resampleArcLength(perim, 512);

  const bowIdx = argmax(P, p => p.x);
  const minX = min(P, p => p.x).x;

  const transomRun = longestLowCurvatureRun(P, { xLessThan: minX + 0.05, kThresh: 0.08 });
  const transomLine = fitLineXZ(transomRun.points);

  const sternCornerL = endpointWithMinZ(projectEndpoints(transomRun, transomLine));
  const sternCornerR = endpointWithMaxZ(projectEndpoints(transomRun, transomLine));

  const bowShoulders = findBowShoulders(P, bowIdx);

  return {
    bow_tip: { ...P[bowIdx], s: sOfIndex(bowIdx) },
    stern_corner_L: { ...sternCornerL, s: sOfPoint(sternCornerL) },
    stern_corner_R: { ...sternCornerR, s: sOfPoint(sternCornerR) },
    bow_shoulder_L: { ...bowShoulders.L, s: sOfPoint(bowShoulders.L) },
    bow_shoulder_R: { ...bowShoulders.R, s: sOfPoint(bowShoulders.R) },
  };
}

function bowFrame(C: (s:number)=>Vec3, s0:number): Frame {
  const eps = 0.01; // meters along arc length
  const p0 = C(s0), pL = C(s0-eps), pR = C(s0+eps);

  let tL = normalize(sub(p0, pL));
  let tR = normalize(sub(pR, p0));
  let fwd = normalize(add(tL, tR));
  if (fwd.x < 0) fwd = mul(fwd, -1);

  const up0 = vec3(0,1,0);
  let right = normalize(cross(up0, fwd));
  if (right.z < 0) right = mul(right, -1);

  const up = cross(fwd, right);
  return { fwd, right, up, origin: p0 };
}

1) What an OrthoSplinePatch is

A patch is a local 2D coordinate space (an ortho view plane) containing:

Edge layers: named spline paths (silhouettes, seams, creases, trims)

Field layers: grayscale / signed fields (depth, height, thickness, blend weights, curvature intent)

Ports: parameterized boundary segments that can connect to other patches (node graph wiring)

Anchors: keypoints (bow tip, transom corners, max beam station, centerline, etc.)

Calibration: px↔meter and frame alignment so it’s metrically true

It outputs both editable vector truth and rasterizable constraint fields.

2) Ortho → Canonical Mapping

The rule: each view edits a disjoint subset by default, and any overlap uses explicit dominance weights.

Dominance defaults

TOP dominates B_halfBeam(s) (width/planform)

SIDE dominates K_keelY(s) and D_deckCLY(s) (heights/profile)

REAR dominates deckCrown.amp_at_stern, transom top edge crown, corner radii

FRONT dominates sectionStyle (V-ness, bilge softness, flare) if used

3) Handle Bindings per View
3.1 TOP view handles

Visible curves: rail seam (and optionally outer silhouette)

(H1) Rail seam spline points → B_halfBeam(s)

When you drag a point on the rail seam:

View plane coords are (u,v) = (x,z)

Convert to station:

s = clamp(x / LOA, 0, 1)

Convert to half-beam:

B_new(s) = abs(z) (with symmetry enforced)

Update rule (station-local):

Find nearest spline knot(s) in B_halfBeam around that s

Apply a local spline edit (knot move or localized delta field)

Run fairness smoothing (2nd-derivative penalty) so you don’t introduce wiggles

This makes TOP editing literally “editing B(s)”.

(H2) Bow tip anchor in TOP → A_BOW_TIP_DECK.x

Dragging bow tip anchor changes:

LOA (if you allow it), or

a local “front extent” constraint used by bow knife + deck overhang

For Laser: I’d keep LOA mostly fixed and treat this as a front keypoint constraint unless you explicitly want scaling.

3.2 SIDE view handles

Visible curves: keel line, deck line, knife line

(H3) Keel line spline → K_keelY(s)

Drag keel curve point at x:

s = x/LOA

K_new(s) = y

(H4) Deck line spline → D_deckCLY(s)

Drag deck curve point at x:

s = x/LOA

D_new(s) = y

Laser default “flat deck in side view” is just:

constrain D_deckCLY to have near-zero slope (or lock it entirely), unless user toggles “allow micro-sheer”.

(H5) Knife edge guide line → bowKnife params

SIDE is where your “cone but clipped” becomes perfectly controllable.

You expose two anchors in SIDE:

knife_upper (near deck seam height)

knife_lower (near keel height)

Dragging those updates one of two modes:

Mode A: knife_len

knife_len = x_upper - x_lower projected along your definition
Mode B: knife_angle

knife_angle = atan2(Δy, Δx) (then clamp)

And upper_edge_bias is basically “where along height the clip is most forward.”

3.3 REAR view handles

Visible curves: deck crown arc, transom outline corners

REAR plane coords are (u,v) = (z,y) at x = 0 (transom station).

(H6) Deck rear crown curve → deckCrown.amp_at_stern (+ optional width shape)

crown amplitude = peak(y at z=0) − y at z=±B(0)

curve shape can tune profile_exp if you expose it

(H7) Transom top edge crown → transom.top_edge_crown_amp

This is the gentle curve where the deck meets transom at the seam height.

(H8) Corner radii handles

top corner radius → transom.top_corner_radius (square intent but rounded)

bottom corner radius → transom.bottom_corner_radius (quadrant-cylinder feel)

3.4 FRONT view handles (optional but powerful)

FRONT plane (z,y) at a chosen station s_current.

(H9) Cross-section curve editor → sectionStyle params

You don’t store the whole section curve; you store the few-parameter family:

pV (keel sharpness)

pS (side convexity)

bilge_center_xi, bilge_softness

flare

This is exactly where your “curved triangle prism” becomes directly sculptable.

4) The No-Drift Update Algorithm

Every edit is converted into a canonical delta and then re-projected.

4.1 Canonicalization step

Given a handle move in view space, compute a candidate update:

ΔB(s) or ΔK(s) or ΔD(s) etc.

4.2 Constraint solve (tiny, local, fast)

Apply a local solve that enforces:

symmetry

fairness (no wiggle)

anchor consistency (bow tip / transom plane)

dominance (TOP width, SIDE height)

4.3 Rebuild

Recompile in fixed order:

Base hull (from B/K/D + sectionStyle + seam)

Deck (from D + seam + deckCrown)

Bow knife macro

Rail lip seam op

4.4 Reproject + overlay errors

Reproject canonical curves back into all views:

show ghost curves (what other views imply)

highlight mismatch zones (edge vs edge, or edge vs field)

No hidden disagreement: if SIDE and TOP imply incompatible seam geometry, it appears as an explicit violation.

8) Minimal pseudocode
function applyHandleDrag(edit) {
  const p = state.params;

  // 1) Build observation deltas (only from the edited view/handle)
  const obs = buildObservationsFromEdit(edit);

  // 2) Build weights with local dominance window
  const W = buildWeights({
    activeView: edit.view,
    activeFamily: edit.family,   // B/K/D/knife/etc
    s0: edit.s0,
    sigma: state.ui.localSigma
  });

  // 3) Assemble residual r(p) and Jacobian J in a reduced subspace
  const subspace = pickLocalSubspace(p, edit);
  const { r, J } = assembleLeastSquares(p, obs, subspace);

  // 4) Add fairness and damping
  const L = assembleFairness(subspace);
  const H = J.T * W*W * J + lamFair * L.T*L + lamDamp * I;
  const g = J.T * W*W * r;

  // 5) Solve H Δ = -g
  const d = solveSparse(H, -g);

  // 6) Apply update, enforce hard constraints by construction
  state.params = applyUpdate(p, subspace, d);
  state.params = enforceByParametrization(state.params);

  // 7) Recompile + reproject ghosts + mismatch overlays
  recompileModel(state.params);
  reprojectAllViews(state.params);
  updateDiagnostics();
}
